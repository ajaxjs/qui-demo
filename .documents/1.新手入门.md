# 配置

配置文件：`qui.config.js`

```javascript
import LocalStorage from '@/uni_modules/kv-qui/plugins/LocalStorage.js'
export default {
	// UI 
	config: {
		notify: {
			position: 'top',
			timeout: 2500
		},
		// 系统主题，不设置则获取
		dark: false,
		// 默认 web-view 路径，$uni.openUrl({href:''})
		webViewPath: '/pages/index/broswer'
	},
	iconSet: {
		icon:'material-icons',
		field: {
			error: 'error',
			clear: 'cancel',
		}
	}, // icon set
	lang: 'zh-CN',
	// plugins 插件
	plugins: [
		LocalStorage,
	],
}
```
